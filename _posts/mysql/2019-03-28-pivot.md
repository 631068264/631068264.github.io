CREATE TABLE tb (
	name VARCHAR ( 10 ),
	course VARCHAR ( 10 ),
	score INT
);

```sql
SELECT
	name AS name,
	max( CASE course WHEN '语文' THEN score ELSE 0 END ) 语文,
	max( CASE course WHEN '数学' THEN score ELSE 0 END ) 数学,
	max( CASE course WHEN '物理' THEN score ELSE 0 END ) 物理 
FROM
	tb
GROUP BY
	name
```
